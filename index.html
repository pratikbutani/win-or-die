    <script>

        let ballsLeft, maxPick, cycle, gameActive = false;


        function startGame() {

            ballsLeft = parseInt(document.getElementById('total-balls').value);

            maxPick = parseInt(document.getElementById('max-pick').value);

            cycle = maxPick + 1;

            gameActive = true;


            document.getElementById('setup').style.display = 'none';

            document.getElementById('game').style.display = 'flex';

            updateBallsLeft();

            document.getElementById('message').textContent = 'Your turn, Pratik!';

            createPlayerControls();

            document.getElementById('robot').classList.add('jumping');

        }


        function updateBallsLeft() {

            document.getElementById('balls-left').textContent = `${ballsLeft} balls left`;

        }


        function createPlayerControls() {

            const controls = document.getElementById('player-controls');

            controls.innerHTML = '';

            for (let i = 1; i <= Math.min(maxPick, ballsLeft); i++) {

                const btn = document.createElement('button');

                btn.textContent = `Pick ${i}`;

                btn.onclick = () => playerMove(i);

                controls.appendChild(btn);

            }

        }


        function playerMove(num) {

            if (!gameActive || num > ballsLeft) return;


            ballsLeft -= num;

            document.getElementById('player').classList.remove('jumping');

            document.getElementById('player').classList.add('player-jump');

            document.getElementById('basket').classList.add('bounce');

            document.getElementById('message').textContent = `Pratik picked ${num} balls!`;

            updateBallsLeft();

            document.getElementById('player-controls').innerHTML = ''; // Disable buttons during animation


            setTimeout(() => {

                document.getElementById('player').classList.remove('player-jump');

                document.getElementById('basket').classList.remove('bounce');

                if (checkWin()) return; // Stop if game ends

                robotMove();

            }, 1000);

        }


        function robotMove() {

            if (!gameActive) return;


            document.getElementById('robot').classList.remove('jumping');

            document.getElementById('player').classList.add('jumping');

            const remainder = ballsLeft % cycle;

            const robotPick = remainder === 0 ? 1 : remainder;

            ballsLeft -= robotPick;


            document.getElementById('robot').classList.add('robot-jump');

            document.getElementById('basket').classList.add('bounce');

            document.getElementById('message').textContent = `Robot picked ${robotPick} balls!`;

            updateBallsLeft();


            setTimeout(() => {

                document.getElementById('robot').classList.remove('robot-jump');

                document.getElementById('basket').classList.remove('bounce');

                if (checkWin()) return; // Stop if game ends

                document.getElementById('message').textContent = 'Your turn, Pratik!';

                createPlayerControls();

            }, 1000);

        }


        function checkWin() {

            if (ballsLeft <= 0) {

                gameActive = false;

                const winner = ballsLeft === 0 ? 'Pratik' : 'Robot';

                document.getElementById('message').textContent = `${winner} wins!`;

                document.getElementById('player-controls').innerHTML = '';

                document.getElementById('player').classList.remove('jumping');

                document.getElementById('robot').classList.remove('jumping');

                return true;

            }

            return false;

        }
